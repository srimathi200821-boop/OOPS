import javax.swing.*; 
import java.awt.*; 
class TimerTask implements Runnable { 
private int seconds; 
private boolean running = false; 
private boolean paused = false; 
private JLabel label; 
public TimerTask(int seconds, JLabel label) { 
this.seconds = seconds; 
this.label = label; 
} 
    public void start() { 
        if (!running) { 
            running = true; 
            new Thread(this).start(); 
        } 
    } 
    public void pause() { 
        paused = true; 
    } 
    public void resume() { 
        paused = false; 
    } 
    @Override 
    public void run() { 
        while (running && seconds >= 0) { 
            try { 
                Thread.sleep(1000); 
                if (!paused) { 
                    SwingUtilities.invokeLater(() -> 
                            label.setText("Time: " + seconds)); 
                    seconds--; 
                } 
            } catch (InterruptedException e) { 
                e.printStackTrace(); 
            } 
        } 
        running = false; 
    } 
} 
public class MultiTimerApp { 
    public static void main(String[] args) { 
        JFrame frame = new JFrame("Multi Timer App"); 
        frame.setSize(450, 200); 
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE); 
        frame.setLayout(new FlowLayout()); 
        JLabel inputLabel = new JLabel("Enter Seconds:"); 
        JTextField inputField = new JTextField(10); 
        JButton startBtn = new JButton("Start Timer"); 
        JButton pauseBtn = new JButton("Pause"); 
        JButton resumeBtn = new JButton("Resume"); 
        JLabel timeLabel = new JLabel("Time: 0"); 
        timeLabel.setFont(new Font("Arial", Font.BOLD, 16)); 
        final TimerTask[] timer = new TimerTask[1]; 
        startBtn.addActionListener(e -> { 
            try { 
                int sec = Integer.parseInt(inputField.getText()); 
                timer[0] = new TimerTask(sec, timeLabel); 
                timer[0].start(); 
            } catch (Exception ex) { 
                JOptionPane.showMessageDialog(frame, "Enter valid number!"); 
            } 
        }); 
     pauseBtn.addActionListener(e -> { 
            if (timer[0] != null) timer[0].pause(); 
       }); 
        resumeBtn.addActionListener(e -> { 
            if (timer[0] != null) timer[0].resume(); 
        }); 
        frame.add(inputLabel); 
        frame.add(inputField); 
        frame.add(startBtn); 
        frame.add(timeLabel); 
        frame.add(pauseBtn); 
        frame.add(resumeBtn); 
      frame.setVisible(true); 
    } 
}
