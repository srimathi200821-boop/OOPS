import java.util.*;
import java.util.Scanner;
abstract class  TemperatureControl{
   protected  double temperature;
   TemperatureControl(double temperature){
       this.temperature=temperature;
   }
   abstract void increase(double value);
   abstract void decrease(double value);
   void showTemperature() {
       System.out.println("Current Temperature: " +temperature +"°C");
}
}
class RoomTemperature  extends TemperatureControl{
     RoomTemperature(double temperature){
         super(temperature);
     }
     void increase(double value){
         temperature +=value;
     }
     void decrease(double value){
         temperature -=value;
     }
}


public class Solution {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.println("Enter initial temperature:");
        double temp = sc.nextDouble();

        System.out.println("Enter value to increase temperature:");
        double inc = sc.nextDouble();

        System.out.println("Enter value to decrease temperature:");
        double dec = sc.nextDouble();

        TemperatureControl t;
        t = new RoomTemperature(temp);

        t.increase(inc);
        t.decrease(dec);
        t.showTemperature();
    }
}
